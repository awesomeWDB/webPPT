//user.js
const user = {
  state: {
    token: getToken(),
    uMobile: '',
    code: '', //code码
    openId: '',
    yanzhengma: '', //验证码
    name: '', //姓名
    role: '', //角色
    hospital: '', //医院
    department: '' //科室
  },

  mutations: {
    SET_TOKEN: (state, token) => {
      state.token = token
    },
    SET_USER: (state, params) => {
      for (let key in params) {
        if (params[key]) {
          console.log(key, params[key])
          state[key] = params[key]
        }
      }
    },
  },

  actions: {
    //
    ValidateCode({commit}, code) {
      return new Promise((resolve, reject) => {
        validateCode({
          code
        }).then(res => {
          commit('SET_USER', {
            openId: res.data
          })
          resolve(res)
        }).catch(err => {
          reject(err)
        })
      })
    },
    //
    ValidateOpenId({commit}, openid) {
      return new Promise((resolve, reject) => {
        validateOpenId({
          openid
        }).then(res => {
          if (res.code == 200) {
            commit('SET_USER', {
              uMobile: res.data.uMobile,
              name: res.data.uName, //姓名
              role: res.data.rId, //角色
              hospital: res.data.cRealname, //医院
              department: res.data.uDepartment //科室
            })
            resolve(res)
          }else{
            resolve(res)
          }
        }).catch(err => {
          reject(err)
        })
      })
    }
  }
}
//option.js
const option = {
  state: {
    projects: [],
    productType: constOptions.productType,
    products: [],
    sampleType: [],
    hospitals: []
  },

  mutations: {
    SET_PROJECTS: (state, arr) => {
      state.projects = arr
    },
    SET_PRODUCTS: (state, arr) => {
      state.products = arr
    },
    SET_SAMPLETYPE: (state, arr) => {
      state.sampleType = arr
    },
    SET_HOSPITALS: (state, arr) => {
      state.hospitals = arr
    },
  },

  actions: {
    SetProjects({ commit }, ) {
      return new Promise((resolve, reject) => {
        getProject().then(res => {
          commit('SET_PROJECTS', res.data)
          resolve(res)
        }).catch(err => {
          reject(err)
        })
      })
    },
    SetProducts({ commit }, pType) {//pType产品大类 236=臻会选 243=贝安臻
      return new Promise((resolve, reject) => {
        getProduct(pType).then(res => {
          commit('SET_PRODUCTS', res.data)
          resolve(res)
        }).catch(err => {
          reject(err)
        })
      })
    },
    SetSampleType({ commit }, params) {
      return new Promise((resolve, reject) => {
        getSampleType(params).then(res => {
          commit('SET_SAMPLETYPE', res.data)
          resolve(res)
        }).catch(err => {
          reject(err)
        })
      })
    },
    SetHospitals({ commit }, params) {
      return new Promise((resolve, reject) => {
        getHosipital(params).then(res => {
          commit('SET_HOSPITALS', res.data)
          resolve(res)
        }).catch(err => {
          reject(err)
        })
      })
    },
  }
}

export default option

axios封装


// 创建axios实例
const service = axios.create({
  baseURL: 'http://192.168.10.77:8100', // api的base_url
  timeout: 10000 // 请求超时时间
})
// request拦截器
service.interceptors.request.use(config => {
  config.headers['Content-Type'] = 'application/x-www-form-urlencoded'
  // 解决axios post请求头为json，方便后台获取数据
  if (config.method === 'post') {
    config.headers['Content-Type'] = 'application/json' //  注意：设置很关键
  }
  // if (store.getters.token) {
  //   config.headers['X-Token'] = getToken() // 让每个请求携带自定义token 请根据实际情况自行修改
  // }
  return config
},
// respone拦截器
service.interceptors.response.use(
  response => {
    //需要权限、重定向之类的可以在这里编写相关逻辑代码
    return response.data
  },
  error => {
    console.log('err' + error)// for debug
    return Promise.reject(error)
  }
)
